library(tidyverse)
inc_data <- function(data){
  if (length(data) == 0) {
    out_data <- rnorm(n = 1, mean = 0, sd = 1)
  } else {
    out_data <- c(data, rnorm(n = 1, mean = 0, sd = 1))
  }
}
